# Сканирование истории при старте (true/false, по умолчанию false)
# Если true - бот попытается просканировать доступную историю сообщений при первом запуске (если БД пустая)
# Если false - сканирование не запускается, новые сообщения будут сохраняться автоматически
# Для ручного сканирования используйте команду /scan_history (только для владельца)
SCAN_HISTORY_ON_START=false

# Платный вход в группу (опционально)
# PAYWALL_ENABLED=true
# MONETIZED_CHAT_ID=-100xxxxxxxxxx
# MONETIZED_CHAT_INVITE_URL=   # опционально: запасная ссылка, если createChatInviteLink недоступен (иначе ссылка создаётся ботом и не протухает)
# MONETIZED_INVITE_CREATES_JOIN_REQUEST=true   # false — если в группе обычный вход без заявок (по умолчанию true = ссылка «на заявку»)
#
# Способы оплаты (можно комбинировать): пользователю хватит оплатить ОДНИМ из включённых.
#
# A) ЮKassa (карта, ссылка в ЛС) — задай RUB/сумму и ключи:
# PAYMENT_CURRENCY=RUB
# PAYMENT_AMOUNT_RUB=100
# PAYMENT_AMOUNT_MINOR_UNITS=10000
# YOOKASSA_SHOP_ID=...
# YOOKASSA_SECRET_KEY=...
#
# B) Дополнительно Telegram Stars при PAYMENT_CURRENCY=RUB + ЮKassa (рекомендуемый «два способа» без XTR):
# PAYMENT_STARS_ENABLED=true
# PAYMENT_STARS_AMOUNT=100
#
# C) Оплата в Telegram картой через провайдера @BotFather (отдельно или вместе со звёздами/ЮKassa):
# PAYMENT_PROVIDER_TOKEN=...
# PAYMENT_CURRENCY=RUB
#
# D) Звёзды (XTR) и одновременно ЮKassa в рублях — задай сумму карты отдельно (основная PAYMENT_CURRENCY=XTR не задаёт рубли для ЮKassa):
# PAYMENT_CURRENCY=XTR
# PAYMENT_STARS_AMOUNT=100
# PAYMENT_AMOUNT_RUB=100
# (или PAYMENT_YOOKASSA_AMOUNT_RUB / PAYMENT_YOOKASSA_AMOUNT_MINOR_UNITS — только для ЮKassa)
# + YOOKASSA_SHOP_ID / YOOKASSA_SECRET_KEY — тогда в боте две кнопки: звёзды и ссылка ЮKassa.
#
# E) Вебхук ЮKassa (ms_payments): бот создаёт платёж и шлёт ссылку; для автозачёта нужен POST на сервис вебхука.
# Прямой URL вебхука для каждого платежа (API ЮKassa notification_url). Надёжнее, чем только ЛК:
# YOOKASSA_NOTIFICATION_URL=https://<публичный-ms_payments>/api/v1/webhook/payment
# Если вебхук не доходит: после оплаты нажми /start — бот опрашивает GET /v3/payments/{id} (нужен yookassa_payment_id в БД, пишется при новой ссылке).
# YOOKASSA_WEBHOOK_VERIFY_BASIC_AUTH=false
# true — только если в ЛК ЮKassa для URL уведомлений включена HTTP Basic (логин shopId, пароль secret_key)
# YOOKASSA_RETURN_URL=https://t.me/YourBot  # куда вернуть пользователя после оплаты (https), по умолчанию можно MONETIZED_CHAT_INVITE_URL
# В кабинете ЮKassa URL уведомлений: https://<домен>/api/v1/webhook/payment (рекомендуется; как в папке api/v1)
# Допустим также: https://<домен>/webhook/payment — тот же код, без префикса.
# В логах должны быть app.webhook HTTP POST …/webhook/payment (с /api/v1 или без).
#
# ms_payments (Railway / payment-webhook):
# На Railway создай ОТДЕЛЬНЫЙ сервис с Dockerfile.ms_payments; публичный URL этого сервиса — в настройках ЮKassa.
# Открой https://<этот-домен>/ в браузере: JSON с "service":"ms_payments". Если 404/другое — в ЮKassa указан URL бота, не вебхука.
# DATABASE_URL — скопируй ТУ ЖЕ строку, что у бота (ms_leo). По ней вебхук обновляет paywall_access_requests;
# иначе оплата в ЮKassa есть, а ms_leo не поймёт, что доступ куплен.
# PAYMENT_DATABASE_URL — опционально: вторая БД только для таблицы yookassa_payment_events (история вебхуков).
# На доступ в группу влияет только DATABASE_URL.
# PAYMENT_DATABASE_URL=postgresql://postgres:password@localhost:5432/leo_payments?sslmode=disable
#
# ms_payments: повторы подключения к Postgres при старте (другой контейнер / Railway).

# DB_CONNECT_RETRY_DELAY_SEC=2
