# Telegram Bot Token
# Получите токен у @BotFather в Telegram
API_TOKEN=your_bot_token_here

# Owner ID (ваш Telegram ID)
OWNER_ID=your_telegram_id_here

# PostgreSQL Database Configuration
# ВАЖНО: Используйте формат postgresql:// (не postgresql+asyncpg://)
# Этот бот использует драйвер lib/pq, который требует формат postgresql://

DATABASE_URL=postgresql://username:password@localhost:5432/leo_bot_db?sslmode=disable

# Пример для локальной разработки:
# DATABASE_URL=postgresql://postgres:password@localhost:5432/leo_bot_db?sslmode=disable

# Пример для Docker (postgres - имя сервиса в docker-compose):
# DATABASE_URL=postgresql://postgres:password@postgres:5432/leo_bot_db?sslmode=disable

# Пример для Railway:
# DATABASE_URL=postgresql://postgres:password@containers-us-west-1.railway.app:5432/railway?sslmode=disable

# OpenRouter API Key (для ИИ функций)
# Получите ключ на https://openrouter.ai
OPENROUTER_API_KEY=your_openrouter_api_key_here

# Инструкция:
# 1. Скопируйте этот файл и переименуйте в .env
# 2. Замените your_bot_token_here на ваш реальный токен бота
# 3. Замените your_telegram_id_here на ваш Telegram ID
# 4. Настройте DATABASE_URL для вашей PostgreSQL базы данных
# 5. Замените your_openrouter_api_key_here на ваш OpenRouter API ключ (для ИИ функций)
# 6. Файл .env будет автоматически игнорироваться git'ом
