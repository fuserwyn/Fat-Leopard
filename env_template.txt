OWNER_ID=your_telegram_id_here

# PostgreSQL Database Configuration
# ВАЖНО: Используйте формат postgresql:// (не postgresql+asyncpg://)
# Этот бот использует драйвер lib/pq, который требует формат postgresql://

DATABASE_URL=postgresql://username:password@localhost:5432/leo_bot_db?sslmode=disable

# Пример для локальной разработки:
# DATABASE_URL=postgresql://postgres:password@localhost:5432/leo_bot_db?sslmode=disable

# DATABASE_URL=postgresql://postgres:password@postgres:5432/leo_bot_db?sslmode=disable

# Пример для Railway:
# DATABASE_URL=postgresql://postgres:password@containers-us-west-1.railway.app:5432/railway?sslmode=disable

# OpenRouter API Key (для ИИ функций)
# Получите ключ на https://openrouter.ai
OPENROUTER_API_KEY=your_openrouter_api_key_here

# OpenRouter Model (опционально, по умолчанию deepseek/deepseek-r1-0528)
# Примеры моделей:
# - deepseek/deepseek-r1-0528 (рекомендуется)
# - deepseek/deepseek-chat-v3.1:free (бесплатная, требует включения "Enable free endpoints that may train on inputs")
# - deepseek/deepseek-chat (платная, не требует настройки)
# - meta-llama/llama-3.1-8b-instruct:free (бесплатная)
# - google/gemini-flash-1.5-8b:free (бесплатная)
# Список моделей: https://openrouter.ai/models
OPENROUTER_MODEL=deepseek/deepseek-r1-0528

# Сканирование истории при старте (true/false, по умолчанию false)
# Если true - бот попытается просканировать доступную историю сообщений при первом запуске (если БД пустая)
# Если false - сканирование не запускается, новые сообщения будут сохраняться автоматически
# Для ручного сканирования используйте команду /scan_history (только для владельца)
SCAN_HISTORY_ON_START=false

# Платный вход в группу (опционально)
# PAYWALL_ENABLED=true
# MONETIZED_CHAT_ID=-100xxxxxxxxxx
# MONETIZED_CHAT_INVITE_URL=   # опционально: запасная ссылка, если createChatInviteLink недоступен (иначе ссылка создаётся ботом и не протухает)
# MONETIZED_INVITE_CREATES_JOIN_REQUEST=true   # false — если в группе обычный вход без заявок (по умолчанию true = ссылка «на заявку»)
#
# Способ оплаты для пользователей — выбери один (или Telegram, или ЮKassa):
#
# A) Оплата в Telegram: токен провайдера из @BotFather → счёт с кнопкой «Оплатить» в чате
# PAYMENT_PROVIDER_TOKEN=...
# PAYMENT_CURRENCY=RUB
# PAYMENT_AMOUNT_RUB=100
# (для RUB удобнее рубли; копейки можно так: 99.90. Если задано — перекрывает PAYMENT_AMOUNT_MINOR_UNITS)
# PAYMENT_AMOUNT_MINOR_UNITS=10000
#
# B) Оплата картой ЮKassa: бот сам создаёт платёж и шлёт ссылку; нужен публичный вебхук (сервис ms_payments / docker-compose payment-webhook).
# YOOKASSA_SHOP_ID=...
# YOOKASSA_SECRET_KEY=...
# YOOKASSA_WEBHOOK_VERIFY_BASIC_AUTH=false
# true — только если в ЛК ЮKassa для URL уведомлений включена HTTP Basic (логин shopId, пароль secret_key)
# YOOKASSA_RETURN_URL=https://t.me/YourBot  # куда вернуть пользователя после оплаты (https), по умолчанию можно MONETIZED_CHAT_INVITE_URL
# В кабинете ЮKassa URL уведомлений (точный путь): https://<публичный-домен-ms_payments>/webhook/payment
# Railway: сервис ms_payments должен быть с публичным URL/портом; в логах при каждой попытке ЮKassa появятся строки app.webhook HTTP POST /webhook/payment.
# Если в логах только старт uvicorn и нет POST /webhook/payment — уведомления в кабинете не долетают (или путь не тот).
#
# ms_payments (Railway / payment-webhook):
# DATABASE_URL — скопируй ТУ ЖЕ строку, что у бота (ms_leo). По ней вебхук обновляет paywall_access_requests;
# иначе оплата в ЮKassa есть, а ms_leo не поймёт, что доступ куплен.
# PAYMENT_DATABASE_URL — опционально: вторая БД только для таблицы yookassa_payment_events (история вебхуков).
# На доступ в группу влияет только DATABASE_URL.
# PAYMENT_DATABASE_URL=postgresql://postgres:password@localhost:5432/leo_payments?sslmode=disable
#
# ms_payments: повторы подключения к Postgres при старте (другой контейнер / Railway).
# DB_CONNECT_MAX_ATTEMPTS=30
# DB_CONNECT_RETRY_DELAY_SEC=2
