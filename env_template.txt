# Telegram Bot Token (основное имя переменной для Fat Leopard)
# Получите токен у @BotFather в Telegram
FAT_LEOPARD_API_TOKEN=your_bot_token_here
# Если не задано, используется старое имя (совместимость):
# API_TOKEN=your_bot_token_here

# Owner ID (ваш Telegram ID)
OWNER_ID=your_telegram_id_here

# PostgreSQL Database Configuration
# ВАЖНО: Используйте формат postgresql:// (не postgresql+asyncpg://)
# Этот бот использует драйвер lib/pq, который требует формат postgresql://

DATABASE_URL=postgresql://username:password@localhost:5432/leo_bot_db?sslmode=disable

# Пример для локальной разработки:
# DATABASE_URL=postgresql://postgres:password@localhost:5432/leo_bot_db?sslmode=disable

# Пример для Docker (postgres - имя сервиса в docker-compose):
# DATABASE_URL=postgresql://postgres:password@postgres:5432/leo_bot_db?sslmode=disable

# Пример для Railway:
# DATABASE_URL=postgresql://postgres:password@containers-us-west-1.railway.app:5432/railway?sslmode=disable

# OpenRouter API Key (для ИИ функций)
# Получите ключ на https://openrouter.ai
OPENROUTER_API_KEY=your_openrouter_api_key_here

# OpenRouter Model (опционально, по умолчанию deepseek/deepseek-r1-0528)
# Примеры моделей:
# - deepseek/deepseek-r1-0528 (рекомендуется)
# - deepseek/deepseek-chat-v3.1:free (бесплатная, требует включения "Enable free endpoints that may train on inputs")
# - deepseek/deepseek-chat (платная, не требует настройки)
# - meta-llama/llama-3.1-8b-instruct:free (бесплатная)
# - google/gemini-flash-1.5-8b:free (бесплатная)
# Список моделей: https://openrouter.ai/models
OPENROUTER_MODEL=deepseek/deepseek-r1-0528

# Сканирование истории при старте (true/false, по умолчанию false)
# Если true - бот попытается просканировать доступную историю сообщений при первом запуске (если БД пустая)
# Если false - сканирование не запускается, новые сообщения будут сохраняться автоматически
# Для ручного сканирования используйте команду /scan_history (только для владельца)
SCAN_HISTORY_ON_START=false
